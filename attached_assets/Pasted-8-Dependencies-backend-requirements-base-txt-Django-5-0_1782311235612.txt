10. Docker & Compose (backend services only)
# infra/docker-compose.yml — backend-only local dev
version: "3.9"
services:
  postgres:
    image: pgvector/pgvector:pg16
    environment:
      POSTGRES_DB: career_mentor
      POSTGRES_USER: career_mentor
      POSTGRES_PASSWORD: devpassword
    ports: ["5432:5432"]
    volumes: ["pgdata:/var/lib/postgresql/data"]
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U career_mentor"]
      interval: 5s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    ports: ["6379:6379"]

  # Self-hosted Langfuse — fully free, no trace cap. Omit this service and point
  # LANGFUSE_HOST at https://cloud.langfuse.com if you'd rather use their managed free tier.
  langfuse:
    image: langfuse/langfuse:2
    environment:
      DATABASE_URL: postgres://career_mentor:devpassword@postgres:5432/langfuse
      NEXTAUTH_SECRET: devsecret
      NEXTAUTH_URL: http://localhost:3000
      SALT: devsalt
    ports: ["3000:3000"]
    depends_on: [postgres]

  backend:
    build: ../backend
    env_file: ../backend/.env
    depends_on: [postgres, redis]
    ports: ["8000:8000"]
    volumes: ["../backend:/app"]

  celery_worker:
    build: ../backend
    command: celery -A config worker -l info
    env_file: ../backend/.env
    depends_on: [postgres, redis]

  ai_service:
    build: ../ai_service
    env_file: ../ai_service/.env
    depends_on: [postgres, redis, langfuse]
    ports: ["8001:8001"]
    volumes: ["../ai_service:/app"]

volumes:
  pgdata:
For production, drop the bind-mounted volumes, use pinned image tags, and keep ai_service off any publicly published port — identical reasoning to the original spec.

backend/Dockerfile
FROM python:3.12-slim
WORKDIR /app
ENV PYTHONDONTWRITEBYTECODE=1 PYTHONUNBUFFERED=1
RUN apt-get update && apt-get install -y --no-install-recommends libpq-dev gcc curl \
    && rm -rf /var/lib/apt/lists/*
COPY requirements/prod.txt .
RUN pip install --no-cache-dir -r prod.txt
COPY . .
RUN chmod +x entrypoint.sh
EXPOSE 8000
ENTRYPOINT ["./entrypoint.sh"]
CMD ["gunicorn", "config.wsgi:application", "--bind", "0.0.0.0:8000", "--workers", "3"]
