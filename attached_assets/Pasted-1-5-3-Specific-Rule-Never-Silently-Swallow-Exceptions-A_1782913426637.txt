What "real" looks like:

1. Resume file → actually parsed (PDF/DOCX text extraction — pdfplumber / python-docx /
   mammoth, NOT a hardcoded "extracted_text" string)
2. Extracted text → sent to the AI model as actual context, not summarized into a
   generic template before reaching the model
3. AI prompt includes the FULL resume text (or a real, faithful summary, not a

4. AI response is parsed into structured fields (skills, experience years, gaps,
   ATS issues) and persisted to the database tied to that specific resume_id
5. Re-fetching GET /api/v1/resumes/{id}/analysis/ after a refresh returns the
   SAME analysis that was generated — not a freshly re-randomized one

import pdfplumber
from docx import Document as DocxDocument

def extract_text(file_bytes: bytes, content_type: str) -> str:
    """Real text extraction — must actually read the file, not return a stub."""
    if "pdf" in content_type:
        text_chunks = []
        with pdfplumber.open(io.BytesIO(file_bytes)) as pdf:
            for page in pdf.pages:
                page_text = page.extract_text()
                if page_text:
                    text_chunks.append(page_text)
        text = "\n".join(text_chunks)
    elif "wordprocessingml" in content_type or "docx" in content_type:
        doc = DocxDocument(io.BytesIO(file_bytes))
        text = "\n".join(p.text for p in doc.paragraphs if p.text.strip())
    else:
        raise ValueError(f"Unsupported file type: {content_type}")

    if len(text.strip()) < 50:
        # A "successfully parsed" empty/near-empty result is itself a dummy outcome —
        # surface it as a real failure so the user can fix their upload.
        raise ResumeParsingError(
            "Could not extract readable text from this file. "
            "It may be a scanned image — try a text-based PDF or DOCX."
        )
    return text
