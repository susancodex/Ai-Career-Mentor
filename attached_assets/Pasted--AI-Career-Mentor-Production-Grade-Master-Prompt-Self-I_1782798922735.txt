## 0. Non-Negotiable Rules — Read Before Writing a Single Line

These are not suggestions. Violations silently break production.

1. **Inspect before you touch.** Before editing any file, read it. Before wiring any integration,
   trace the full call chain in your head and identify every place it can silently fail.

2. **Self-heal broken features automatically.** As you work through each section, run the
   verification steps. If a verification fails, fix it immediately — do not leave a comment,
   do not add a TODO, do not move on. A broken feature with a TODO comment is just a broken
   feature with extra decoration.

3. **Follow STOP checkpoints in order.** Each one is a real gate. A green container health
   check does not mean the feature works. Run the actual curl or browser test specified.

4. **No placeholder code anywhere in the codebase.** Search for: `TODO`, `FIXME`, `pass`,

   scroll on any screen.

   Browser → Nginx → Django → Celery → AI Service → Gemini → Redis → Browser SSE
   must be verified with a real streaming call before you build anything else on top of it.

7. **Any real secret found in the codebase means stop immediately.** Rotate the key,

### 2.1 Authentication System

**Required behaviour (inspect and fix each item):**

- `POST /api/v1/auth/register/` → 201, `{ access }` in body, `refresh_token` httpOnly cookie set
- `POST /api/v1/auth/login/` → 200, same shape
- `POST /api/v1/auth/refresh/` → reads cookie only, never body or Authorization header → 200 `{ access }`
- `POST /api/v1/auth/logout/` → 204, clears cookie with `Max-Age=0`
- `POST /api/v1/auth/password/forgot/` → accepts `{ email }`, sends reset email, always 200
  (never leaks whether the address exists)
- `POST /api/v1/auth/password/reset/` → accepts `{ token, new_password }`, validates token
  expiry (1 hour), invalidates token after first use
- `POST /api/v1/auth/password/change/` → authenticated, accepts `{ current_password, new_password }`,
  re-hashes and saves
- Social auth (Google OAuth) → if configured, token exchange works; if not configured,
  the button is hidden — it does not show a broken flow
- Cookie attributes in production: `httponly=True, secure=True, samesite="None", path="/api/v1/auth/refresh/"`
- Cookie attributes in development: `httponly=True, secure=False, samesite="Lax"`
- `CORS_ALLOW_CREDENTIALS = True` is set
- `CORS_ALLOWED_ORIGINS` is read from env, not hardcoded

```typescript
api.interceptors.response.use(
  (res) => res,
  async (error) => {
    const original = error.config;
    if (error.response?.status === 401 && !original._retry) {
      original._retry = true;
      if (isRefreshing) {
        return new Promise((resolve) => {
          refreshQueue.push((token) => {
            original.headers.Authorization = `Bearer ${token}`;
            resolve(api(original));
          });
        });
      }
      isRefreshing = true;
      try {
        const { data } = await api.post("/auth/refresh/");
        setToken(data.access);
        original.headers.Authorization = `Bearer ${data.access}`;
        refreshQueue.forEach((cb) => cb(data.access));
        refreshQueue = [];
        return api(original);
      } finally {
        isRefreshing = false;
      }
    }
    return Promise.reject(error);
  }
);
```

```python
class AvatarUploadView(APIView):
    """
    Accepts a Cloudinary URL already uploaded by the frontend.
    Validates the URL belongs to our Cloudinary cloud before saving.
    If the user had a previous avatar, deletes it from Cloudinary.
    """
    permission_classes = [IsAuthenticated]

    def post(self, request):
        serializer = AvatarUploadSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)

        cloud_name = settings.CLOUDINARY_CLOUD_NAME
        url        = serializer.validated_data["cloudinary_url"]
        public_id  = serializer.validated_data["cloudinary_public_id"]

        # Validate URL belongs to our cloud
        if f"res.cloudinary.com/{cloud_name}" not in url:
            return Response({"error": "Invalid Cloudinary URL."}, status=400)

        profile = request.user.profile

        # Delete previous avatar from Cloudinary if one exists
        if profile.avatar_public_id:
            try:
                import cloudinary.uploader
                cloudinary.uploader.destroy(profile.avatar_public_id)
            except Exception:
                pass  # Don't fail the upload if cleanup fails

        profile.avatar_url       = url
        profile.avatar_public_id = public_id
        profile.save(update_fields=["avatar_url", "avatar_public_id", "updated_at"])
        return Response({"avatar_url": url})

class DeleteAccountView(APIView):
    permission_classes = [IsAuthenticated]

    def delete(self, request):
        password = request.data.get("password", "")
        if not request.user.check_password(password):
            return Response({"error": "Incorrect password."}, status=400)

        # Clean up Cloudinary avatar if exists
        if hasattr(request.user, "profile") and request.user.profile.avatar_public_id:
            try:
                import cloudinary.uploader
                cloudinary.uploader.destroy(request.user.profile.avatar_public_id)
            except Exception:
                pass

        request.user.delete()
        response = Response(status=204)
        response.delete_cookie("refresh_token")
        return response
```

```typescript
// frontend/src/components/profile/AvatarUpload.tsx
import { useRef, useState } from "react";

const MAX_SIZE_BYTES = 2 * 1024 * 1024; // 2 MB for avatars
const ALLOWED_TYPES  = ["image/jpeg", "image/png", "image/webp", "image/gif"];

export function AvatarUpload({ currentUrl, onSuccess }: Props) {
  const inputRef   = useRef<HTMLInputElement>(null);
  const [preview, setPreview]   = useState<string | null>(null);
  const [loading, setLoading]   = useState(false);
  const [error, setError]       = useState<string | null>(null);

  const handleFile = async (file: File) => {
    setError(null);

    if (!ALLOWED_TYPES.includes(file.type)) {
      setError("Only JPEG, PNG, WebP, and GIF images are accepted.");
      return;
    }
    if (file.size > MAX_SIZE_BYTES) {
      setError("Image must be under 2 MB.");
      return;
    }

    // Show local preview immediately
    setPreview(URL.createObjectURL(file));
    setLoading(true);

    try {
      // Upload to Cloudinary unsigned preset (avatars only)
      const formData = new FormData();
      formData.append("file", file);
      formData.append("upload_preset", import.meta.env.VITE_CLOUDINARY_AVATAR_PRESET);
      formData.append("folder", "avatars");

      const cloudRes = await fetch(
        `https://api.cloudinary.com/v1_1/${import.meta.env.VITE_CLOUDINARY_CLOUD_NAME}/image/upload`,
        { method: "POST", body: formData }
      );
      if (!cloudRes.ok) throw new Error("Cloudinary upload failed.");
      const { secure_url, public_id } = await cloudRes.json();

      // Save to our backend
      await api.post("/me/avatar/", {
        cloudinary_url: secure_url,
        cloudinary_public_id: public_id,
      });

      onSuccess(secure_url);
    } catch (e: any) {
      setError(e.message ?? "Upload failed. Try again.");
      setPreview(null);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div>
      {/* Avatar display — 96px circle, touch-friendly */}
      <button
        onClick={() => inputRef.current?.click()}
        aria-label="Change profile picture"
        style={{ width: 96, height: 96, borderRadius: "50%", overflow: "hidden",
                 border: "2px solid #e2e8f0", cursor: "pointer", background: "#f1f5f9" }}
      >
        {(preview || currentUrl) ? (
          <img src={preview ?? currentUrl} alt="Your avatar"
               style={{ width: "100%", height: "100%", objectFit: "cover" }} />
        ) : (
          <span style={{ fontSize: 36 }}>👤</span>
        )}
      </button>

      <input ref={inputRef} type="file" accept={ALLOWED_TYPES.join(",")}
             style={{ display: "none" }}
             onChange={(e) => { const f = e.target.files?.[0]; if (f) handleFile(f); }} />

      {loading && <p>Uploading…</p>}
      {error   && <p role="alert" style={{ color: "red" }}>{error}</p>}
    </div>
  );
}
```

### 2.4 AI Chat — Streaming (Most Critical Feature)

The full call chain (read it twice):

```
Browser EventSource
  → POST /api/v1/chat/sessions/{id}/messages/        [Nginx: no buffering]
      → Django view: persists message, fires Celery task, subscribes to Redis
          → Celery: signs HMAC, calls POST http://ai_service:8001/chat/stream/
              → FastAPI: verifies HMAC, invokes LangGraph → Gemini
              → Streams SSE tokens back to Celery
          → Celery: publishes each token to Redis channel
      → Django SSE view: listens on Redis, yields tokens to browser
  → Browser: renders tokens progressively, word by word
```

**HMAC signing (inspect and fix):**

```python
# backend/apps/chat/tasks.py
import hashlib, hmac, time, httpx, json, redis
from django.conf import settings

def _hmac_headers(body: bytes) -> dict:
    timestamp = str(int(time.time()))
    sig = hmac.new(
        settings.AI_SERVICE_SHARED_SECRET.encode(),
        msg=f"{timestamp}.".encode() + body,
        digestmod=hashlib.sha256,
    ).hexdigest()
    # NEVER log these headers
    return {"X-Timestamp": timestamp, "X-Signature": sig, "Content-Type": "application/json"}

@shared_task(bind=True, max_retries=3, default_retry_delay=2)
def stream_chat_message(self, session_id: str, message: str, user_id: int):
    r = redis.from_url(settings.REDIS_URL)
    channel = f"chat:stream:{session_id}"
    payload = json.dumps({"session_id": session_id, "message": message, "user_id": user_id}).encode()
    headers = _hmac_headers(payload)
    try:
        with httpx.stream("POST", f"{settings.AI_SERVICE_URL}/chat/stream/",
                          content=payload, headers=headers, timeout=300) as resp:
            resp.raise_for_status()
            for line in resp.iter_lines():
                if line.startswith("data:"):
                    r.publish(channel, line)
        r.publish(channel, 'data: {"done":true}')
    except Exception as exc:
        r.publish(channel, f'data: {{"error":"{type(exc).__name__}"}}')
        raise self.retry(exc=exc)
```

```python
    stream_chat_message.delay(str(session_id), content, request.user.id)

    r = redis.from_url(settings.REDIS_URL)
    ps = r.pubsub()
    ps.subscribe(f"chat:stream:{session_id}")

    def event_stream():
        try:
            for msg in ps.listen():
                if msg["type"] == "message":
                    data = msg["data"].decode()
                    yield f"{data}\n\n"
                    if '"done"' in data or '"error"' in data:
                        break
        finally:
            ps.unsubscribe()
            ps.close()

    response = StreamingHttpResponse(event_stream(), content_type="text/event-stream")
    response["Cache-Control"] = "no-cache"
    response["X-Accel-Buffering"] = "no"
    return response
```

**FastAPI HMAC middleware (inspect and fix):**

```python
# ai_service/middleware/hmac_auth.py
import hashlib, hmac, os, time
from fastapi import Request, HTTPException

_SECRET = os.environ.get("AI_SERVICE_SHARED_SECRET", "")
if not _SECRET:
    raise RuntimeError("AI_SERVICE_SHARED_SECRET is not set — refusing to start.")

async def verify_hmac(request: Request, call_next):
    ts  = request.headers.get("X-Timestamp", "")
    sig = request.headers.get("X-Signature", "")
    if not ts or not sig:
        raise HTTPException(401, "Missing HMAC headers")
    if abs(time.time() - int(ts)) > 60:
        raise HTTPException(401, "Request expired")
    body = await request.body()
    expected = hmac.new(_SECRET.encode(), f"{ts}.".encode() + body, hashlib.sha256).hexdigest()
    if not hmac.compare_digest(expected, sig):
        raise HTTPException(401, "Invalid signature")
    return await call_next(request)
```

**Frontend chat component — token rendering (inspect and fix):**

```typescript
// frontend/src/components/chat/MessageStream.tsx
export function useChat(sessionId: string) {
  const [messages, setMessages] = useState<Message[]>([]);
  const [streaming, setStreaming] = useState(false);

  const sendMessage = async (content: string) => {
    // Add user message immediately (optimistic UI)
    setMessages(prev => [...prev, { role: "user", content, id: Date.now().toString() }]);
    setStreaming(true);

    // Add empty assistant message placeholder
    const assistantId = (Date.now() + 1).toString();
    setMessages(prev => [...prev, { role: "assistant", content: "", id: assistantId }]);

    // Send and stream
    const res = await fetch(`/api/v1/chat/sessions/${sessionId}/messages/`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${getToken()}`,
      },
      body: JSON.stringify({ content }),
    });

    if (!res.ok || !res.body) {
      setStreaming(false);
      return;
    }

    const reader = res.body.getReader();
    const decoder = new TextDecoder();
    let buffer = "";

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split("\n\n");
      buffer = lines.pop() ?? "";
      for (const line of lines) {
        if (!line.startsWith("data: ")) continue;
        try {
          const parsed = JSON.parse(line.slice(6));
          if (parsed.done) { setStreaming(false); return; }
          if (parsed.error) { setStreaming(false); return; }
          if (parsed.token) {
            setMessages(prev => prev.map(m =>
              m.id === assistantId ? { ...m, content: m.content + parsed.token } : m
            ));
          }
        } catch { /* ignore partial JSON */ }
      }
    }
    setStreaming(false);
  };

  return { messages, sendMessage, streaming };
}
```
